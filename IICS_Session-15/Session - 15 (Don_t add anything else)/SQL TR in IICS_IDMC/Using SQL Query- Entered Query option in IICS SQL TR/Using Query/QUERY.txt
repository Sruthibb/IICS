(
CUSTOMER_ID	NUMBER,
NAME	VARCHAR2(255) ,
ADDRESS	VARCHAR2(255) ,
WEBSITE	VARCHAR2(255) ,
CREDIT_LIMIT	NUMBER(8,2),
CUST_ORDER_ID	NUMBER,
STATUS	VARCHAR2(20) ,
SALESMAN_ID	NUMBER,
ORDER_DATE	DATE
);
