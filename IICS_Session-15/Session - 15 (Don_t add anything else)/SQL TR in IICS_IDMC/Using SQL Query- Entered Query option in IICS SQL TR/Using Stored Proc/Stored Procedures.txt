drop PROCEDURE SP_PROD_AGG;
create or replace PROCEDURE SP_PROD_AGG
(
in_CATEGORY_ID in number,
max_price  out number,
min_price out number,
avg_price out number ,
sum_price out number 
)
as 
Begin 
SELECT MAX(PRICE) , MIN(PRICE) , AVG(PRICE) , SUM(PRICE) into max_price,min_price,avg_price,sum_price
from Prod_src where CATEGORY_ID=in_CATEGORY_ID group by  in_CATEGORY_ID;
end;










drop procedure SP_PROD_AGG_UNC;
CREATE  OR REPLACE 
PROCEDURE SP_PROD_AGG_UNC
(
in_CATEGORY_ID in number,
max_price  out number
)
AS 
Begin
